git status - чекает статус. список изменений
git commit - создает коммит
git merge - слияние
git pull - для загрузки удаленного репозитория
git init - создать репозиторий 
cd .. вернуться назад
git reset - отменяет изменения
git rm - удаляет
git fetch - смотрит есть ли изменения в удаленном репозитории.
git log - показывает все изменения репозитория
git clone - выкачивать репозиторий
